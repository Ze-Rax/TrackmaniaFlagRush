#Include "Libs/Zrx/ModeLibs/FlagRush/FlagRush_Scores.Script.txt" as FlagRush_Scores

Integer Private_DetermineWinnerClan(Integer Clan1Points, Integer Clan2Points) {
	declare Integer Diff = Clan1Points - Clan2Points;
	if(Diff > 0) return 1;
	else if (Diff < 0) return 2;
	return 0;
}

/**
 * Evalutates the current round and determines the winner
 * @return Integer
 * 0 Draw
 * 1,2... for WinnerClan number
 */
Integer EvaluateRound() {
	declare Integer RoundPointsTeam1 = FlagRush_Scores::GetClanRoundPoints(1);
	declare Integer RoundPointsTeam2 = FlagRush_Scores::GetClanRoundPoints(2);
	return Private_DetermineWinnerClan(RoundPointsTeam1, RoundPointsTeam2);
}

/**
 * Evalutates the current map and determines the winner
 * @return Integer
 * 0 for Draw
 * 1,2... for WinnerClan number
 */
Integer EvaluateMap() {
	declare Integer MapPointsTeam1 = FlagRush_Scores::GetClanMapPoints(1);
	declare Integer MapPointsTeam2 = FlagRush_Scores::GetClanMapPoints(2);
	return Private_DetermineWinnerClan(MapPointsTeam1, MapPointsTeam2);
}

/**
 * Evalutates the current match and determines the winner
 * @return Integer
 * 0 for Draw
 * 1,2... for WinnerClan number
 */
Integer EvaluateMatch() {
	declare Integer MatchPointsTeam1 = FlagRush_Scores::GetClanMatchPoints(1);
	declare Integer MatchPointsTeam2 = FlagRush_Scores::GetClanMatchPoints(2);
	return Private_DetermineWinnerClan(MatchPointsTeam1, MatchPointsTeam2);
}
